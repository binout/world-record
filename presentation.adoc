== equals / hasCode

[source, java]
----
public record Point(int x, int y)
----

[source, java]
----
public class Point {
  private final int x;
  private final int y;

  ....

  @Override
  public int hashCode() {
      return Objects.hash(x, y);
  }

  @Override
  public boolean equals(Object obj) {
      if (this == obj) {
          return true;
      } else if (!(obj instanceof Point)) {
          return false;
      } else {
          Point other = (Point) obj;
          return Objects.equals(x, other.x)
            && Objects.equals(y, other.y);
      }
  }
}
